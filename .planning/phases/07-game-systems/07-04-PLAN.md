  pub fn find_glyph_with_distance(&mut self, vector: [f32; 6]) -> (u8, f32) {
      let key = quantize_to_key(&vector);
      if let Some(&(glyph, dist)) = self.cache.get(&key) {
          return (glyph, dist);

      self.cache.put(key, (glyph, dist));
