Implementation (port of C++ game.cpp load sequence):

1. Return early if `assembly.assembled == true`. Separately, return early if `assembly.a3d_handle` is None. These are two distinct guards:
   ```rust
   if assembly.assembled { return; }
   let Some(handle) = assembly.a3d_handle.as_ref() else { return; };
   ```
   NOTE: Bevy systems return `()` -- use `let-else` patterns, NOT the `?` operator.

2. Check if the A3dFile asset is loaded via `a3d_files.get(handle)`.

3. If loaded, resolve sub-asset handles:
   ```rust
   let Some(a3d_file) = a3d_files.get(handle) else { return; };
   let Some(terrain) = terrains.get(&a3d_file.terrain) else { return; };
   let Some(mat_table) = mat_tables.get(&a3d_file.materials) else { return; };
   let Some(world) = worlds.get(&a3d_file.world) else { return; };
   ```

4. When all sub-assets ready:
   - (a) Build RuntimeTerrain: `runtime_terrain.build_from_parsed(terrain_asset)`
   - (b) Build RuntimeWorld: `runtime_world.build_from_parsed(world_asset)`
   - (c) Extract MaterialTable and insert as standalone Bevy Resource:
     ```rust
     let materials: Vec<Material> = convert_material_table(mat_table_asset);
     commands.insert_resource(RuntimeMaterials(materials));
     ```
   - (d) Queue AKM mesh loads using entry API for idempotency:
     ```rust
     mesh_registry.meshes.entry(mesh_name.clone())
         .or_insert_with(|| asset_server.load(&format!("meshes/{}.akm", mesh_name)));
     ```
   - (e) Set `assembly.assembled = true`
   - NOTE: Terrain shadow computation (`update_terrain_dark`) is NOT called here. It runs in Plan 06's own system after assembly completes.

5. Log info about what was assembled (patch count, instance count, material count).

**Complete pipeline implementation with borrow scoping:**

`ensure_buffer_size` takes `&mut SampleBuffer` (exclusive borrow). The field destructure for render stages creates partial borrows. These CANNOT coexist. The resize MUST complete and its borrow must DROP before the destructure begins.

```rust
pub fn render_pipeline_system(/* params as above */) {
    let frame_start = Instant::now();

    // STEP 1: Sync RenderConfig with AsciiCellGrid on window resize.
    // handle_window_resize (output/mod.rs) updates AsciiCellGrid but not RenderConfig.
    if config.ascii_width != cell_grid.width || config.ascii_height != cell_grid.height {
        config.ascii_width = cell_grid.width;
        config.ascii_height = cell_grid.height;
    }

    // STEP 2: Buffer resize (exclusive &mut SampleBuffer borrow).
    ensure_buffer_size(&mut sample_buffer, &config);
    // Borrow DROPPED here.

    // STEP 3: Stage 1 CLEAR
    let t0 = Instant::now();
    sample_buffer.clear();
    timing.clear_us = t0.elapsed().as_micros() as u64;

    // STEP 4: Destructure for render stages (field borrows)
    {
        let buf = &mut sample_buffer.samples;
        let buf_w = sample_buffer.width as i32;   // sample width (e.g. 484)
        let buf_h = sample_buffer.height as i32;   // sample height (e.g. 274)

        // Stage 2: TERRAIN (real rasterization)
        let t1 = Instant::now();
        // If terrain has root and materials are available:
        // Get frustum planes from camera
        // terrain.query_visible(planes, |patch, x, y| {
        //     render_patch(buf, buf_w, buf_h, patch, x, y, &camera.view_tm);
        // })
        timing.terrain_us = t1.elapsed().as_micros() as u64;

        // Stage 3: WORLD (real rasterization for meshes, placeholder for sprites)
        let t2 = Instant::now();
        sprite_queue.clear(); // Phase 5 standalone; Phase 6 moves this to PreUpdate
        // Cast camera pos to f64 for world query:
        // let camera_pos_f64: [f64; 3] = camera.pos.map(|x| x as f64);
        // world_data.query_visible(planes, camera_pos_f64, |visible_inst| {
        //     For Mesh: render_mesh(buf, buf_w, buf_h, mesh_data, &inst_tm, &camera.view_tm)
        //     For Sprite: compute distance, project to screen, push SpriteRenderEntry
        // })
        // NOTE: If mesh not yet loaded in MeshRegistry, skip with warning.
        timing.world_us = t2.elapsed().as_micros() as u64;

        // Stage 4: SHADOW (stub -- deferred to Phase 6)
        let t3 = Instant::now();
        timing.shadow_us = t3.elapsed().as_micros() as u64;

        // Stage 5: REFLECTION (stub -- deferred to Phase 6)
        let t4 = Instant::now();
        timing.reflection_us = t4.elapsed().as_micros() as u64;

    } // mutable borrow of sample_buffer.samples ENDS here

    // STEP 5: Stage 6 RESOLVE (immutable &SampleBuffer borrow)
    let t5 = Instant::now();
    if let Some(mats) = materials.as_ref() {
        let ascii_w = config.ascii_width as usize;
        let ascii_h = config.ascii_height as usize;
        let mut resolve_buf = vec![AnsiCell::default(); ascii_w * ascii_h];
        let mut glyph_sel = AutoMatGlyphSelector::new();
        resolve_to_grid(
            &*sample_buffer,
            &mats.0,
            &mut cell_grid,
            &mut glyph_sel,
            &mut resolve_buf,
        );
    }
    timing.resolve_us = t5.elapsed().as_micros() as u64;

    // STEP 6: Post-RESOLVE Deferred Sprite Blit
    let t6 = Instant::now();
    sprite_queue.sort_far_to_near();
    for entry in sprite_queue.drain() {
        blit_sprite(&mut cell_grid, &entry, &sample_buffer);
    }
    timing.sprite_us = t6.elapsed().as_micros() as u64;
    timing.total_us = frame_start.elapsed().as_micros() as u64;
}
```
