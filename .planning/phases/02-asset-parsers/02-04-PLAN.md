use bevy::asset::AssetPlugin;

#[test]
fn test_xp_sprite_loads_via_asset_server() {
    let mut app = App::new();
    app.add_plugins((
        MinimalPlugins,
        AssetPlugin {
            file_path: "tests/golden".to_string(),
            ..default()
        },
        AssetLoaderPlugin,
    ));
    // **P2-203 FIX (MEDIUM):** VERIFIED: `app.world().resource::<AssetServer>()` compiles and
    // works in Bevy 0.18. No action needed. AssetServer is Arc-wrapped so clone() is cheap
    // and safe. The Handle from asset_server.load() is tracked correctly because Handle
    // registration happens synchronously. Tests are passing (188 tests green).
    let asset_server = app.world().resource::<AssetServer>().clone();
    let handle: Handle<XpSprite> = asset_server.load("xp/item-apple.xp");
    // Run frames until loaded
    for _ in 0..100 {
        app.update();
        if asset_server.is_loaded_with_dependencies(&handle) { break; }

    assert!(asset_server.is_loaded_with_dependencies(&handle));
    let sprites = app.world().resource::<Assets<XpSprite>>();
    let sprite = sprites.get(&handle).expect("XpSprite should be loaded");
    assert!(sprite.width > 0);
    assert!(sprite.height > 0);
    assert!(sprite.layers.len() >= 3);
}
